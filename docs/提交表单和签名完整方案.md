### 1. 上传流程（待实现）

**前端处理**：
```javascript
// 前端流程示例（MaintenanceSubmitPage.vue 增强）

async function uploadFiles(files) {
  const uploadedAttachments = [];

  for (const file of files) {
    // Step 1: 校验文件
    validateFile(file);

    // Step 2: 计算内容哈希
    const contentHash = await calculateFileHash(file);

    // Step 3: 生成缩略图（图片）
    let thumbnailHash = null;
    if (isImageFile(file)) {
      const thumbnail = await generateThumbnail(file);
      thumbnailHash = await calculateFileHash(thumbnail);
    }

    // Step 4: 上传文件到后端
    const uploadResponse = await uploadToBackend(file, {
      contentHash,
      thumbnailHash,
      fileType: getFileType(file)
    });

    // Step 5: 获得存储路径和 attachmentId
    uploadedAttachments.push({
      attachmentId: uploadResponse.attachmentId,
      fileName: file.name,
      mimeType: file.type,
      fileSize: file.size,
      contentHash: uploadResponse.contentHash,
      thumbnailHash: uploadResponse.thumbnailHash,
      storagePath: uploadResponse.storagePath,
      attachmentType: detectAttachmentType(file.type),
      uploadStatus: 'ready'
    });
  }

  return uploadedAttachments;
}

// 计算文件哈希（SHA-256）
async function calculateFileHash(file) {
  const buffer = await file.arrayBuffer();
  const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
  return '0x' + Array.from(new Uint8Array(hashBuffer))
    .map(b => b.toString(16).padStart(2, '0'))
    .join('');
}

// 生成图片缩略图
async function generateThumbnail(file) {
  const canvas = await createImageThumbnail(file);
  return new File([canvas.toBlob()], `thumb_${file.name}`, { type: 'image/jpeg' });
}
```

### 2. 文件类型检测

```javascript
// 前端工具函数
function getFileType(file) {
  const mimeType = file.type;

  // 图片
  if (mimeType.startsWith('image/')) {
    return 'image';
  }

  // 视频
  if (mimeType.startsWith('video/')) {
    return 'video';
  }

  // 文档
  const documentMimes = [
    'application/pdf',
    'application/msword',
    'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
    'application/vnd.ms-excel',
    'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet',
  ];

  if (documentMimes.includes(mimeType)) {
    return 'document';
  }

  return 'other';
}

function detectAttachmentType(mimeType) {
  if (!mimeType) return 'other';

  if (mimeType.startsWith('image/')) return 'image';
  if (mimeType.startsWith('video/')) return 'video';
  if (mimeType.includes('pdf') || mimeType.includes('document') || mimeType.includes('word') || mimeType.includes('sheet')) {
    return 'document';
  }

  return 'other';
}
```

## 哈希生成方案

### 1. 六层哈希结构

| 哈希名称 | 计算内容 | 用途 | 示例 |
|---------|--------|------|------|
| **formHash** | 飞机号、机型、工号、ATA码、工作类型、工卡号、修订号、工作描述、参考文档、发生时间 | 验证基础信息未被篡改 | `0xbf...` |
| **faultHash** | 故障代码、故障描述 | 验证故障信息 | `0x4a...` |
| **partsHash** | 部件列表 (角色、件号、序列号等) | 验证部件清单 | `0x5e...` |
| **measurementsHash** | 测量项目 (测试项、测量值、是否通过) | 验证测量数据 | `0x7c...` |
| **replacementsHash** | 替换清单 (拆下件、装上件等) | 验证替换操作 | `0x9f...` |
| **attachmentManifestHash** | 完整 Manifest JSON | 验证附件清单 | `0x2d...` |
| **signedDigest** | recordId + action + formHash + attachmentManifestHash + signerEmployeeNo | 签名对象 | `0x11...` |

### 2. 哈希算法

**使用 Keccak-256**（与以太坊兼容）：

```javascript
// 标准化为规范 JSON，然后 Keccak-256 哈希
function hashJson(value) {
  const canonical = canonicalJson(value);  // 排序 key、删除空格
  return ethers.keccak256(ethers.toUtf8Bytes(canonical));
}

// 规范 JSON (排序 key，移除空格)
function canonicalJson(value) {
  if (Array.isArray(value)) {
    return `[${value.map(canonicalJson).join(',')}]`;
  }

  if (value && typeof value === 'object') {
    const sortedKeys = Object.keys(value).sort();
    const pairs = sortedKeys.map(key =>
      `${JSON.stringify(key)}:${canonicalJson(value[key])}`
    );
    return `{${pairs.join(',')}}`;
  }

  return JSON.stringify(value);
}
```

### 2. 签名格式（EIP-191）

**前端签名**：
```javascript
// 当前使用 EIP-191 格式
const signedDigest = "0x1234567890abcdef...";  // 32 字节 hex
const message = ethers.getBytes(signedDigest);
const signature = await wallet.signMessage(message);
// 返回: "0x" + 130 字符 (65 字节)
// = "0x" + 32字节r + 32字节s + 1字节v
```

**后端验证**：
```javascript
function ensureSignatureMatchesAddress(signedDigest, signature, expectedAddress) {
  try {
    const recoveredAddress = ethers.verifyMessage(
      ethers.getBytes(signedDigest),
      signature
    );
    return normalizeAddress(recoveredAddress) === normalizeAddress(expectedAddress);
  } catch (error) {
    throw new Error('签名格式不合法或验证失败');
  }
}
```

// 前端后续签名示例
async function handleAppendSignature(recordId, role, action) {
  const token = getAuthToken();

  // 获取签名前上下文
  const prepareResponse = await fetch(
    `/api/maintenance/records/${recordId}/prepare-sign`,
    {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${token}` },
      body: JSON.stringify({ signerRole: role, action })

  const { signedDigest } = await prepareResponse.json();

  // 签名
  const signature = await confirmationWallet.signMessage(
    ethers.getBytes(signedDigest)
  );

  // 提交签名

    `/api/maintenance/records/${recordId}/signatures`,
    {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${token}` },
      body: JSON.stringify({
        signerRole: role,
        action,
        signedDigest,
        signature
      })
    }
  );
}

// 所有字段规范化过程
function normalizeString(value) {
  return String(value || '').trim();
}

function normalizeOptionalString(value) {
  const normalized = normalizeString(value);
  return normalized || null;  // 空字符串转为 null
}

// 数值处理
function assertNonNegativeInteger(value, label, defaultValue = 0) {
  const parsed = value === '' || value == null ? defaultValue : Number(value);
  if (!Number.isInteger(parsed) || parsed < 0) {
    throw new Error(`${label}必须是非负整数`);
  }
  return parsed;
}

// 时间戳处理（链上使用秒级时间戳）

### 3. 性能考虑

- **哈希计算**：前端可缓存哈希值，避免多次计算
- **文件上传**：大文件使用分块上传 + 断点续传
- **签名验证**：后端 cache 最近 N 个签名验证结果
- **数据库**：使用索引加速 recordId、jobCardNo、performerEmployeeNo 查询
